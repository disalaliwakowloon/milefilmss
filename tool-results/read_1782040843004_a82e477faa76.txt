   112→function censorName(name: string): string {
   113→  const chars = name.split('');
   114→  const letterIndices: number[] = [];
   115→  for (let i = 0; i < chars.length; i++) {
   116→    if (chars[i] !== ' ') letterIndices.push(i);
   117→  }
   118→  const numAsterisks = Math.random() < 0.5 ? 4 : 5;
   119→  if (letterIndices.length <= numAsterisks) return name;
   120→  // Shuffle
   121→  for (let i = letterIndices.length - 1; i > 0; i--) {
   122→    const j = Math.floor(Math.random() * (i + 1));
   123→    [letterIndices[i], letterIndices[j]] = [letterIndices[j], letterIndices[i]];
   124→  }
   125→  const selected = letterIndices.slice(0, numAsterisks);
   126→  for (const idx of selected) chars[idx] = '*';
   127→  return chars.join('');
   128→}

   130→function getRandomItem<T>(arr: T[]): T {
   131→  return arr[Math.floor(Math.random() * arr.length)];
   132→}

   134→function getRandomName(): string {
   135→  const name = getRandomItem(ALL_NAMES);
   136→  if (CENSORED_NAMES.includes(name)) return censorName(name);
   137→  return name;

   149→function formatDate(iso: string): string {
   150→  const d = new Date(iso);
   151→  return d.toLocaleDateString('id-ID', { day: 'numeric', month: 'short', year: 'numeric' }) + ' ' + d.toLocaleTimeString('id-ID', { hour: '2-digit', minute: '2-digit' });
   152→}

   154→function formatTimeAgo(dateStr: string): string {
   155→  const now = new Date();
   156→  const lastActive = new Date(dateStr);
   157→  const diffMs = now.getTime() - lastActive.getTime();
   158→  const diffMinutes = Math.floor(diffMs / (1000 * 60));
   159→  const diffHours = Math.floor(diffMs / (1000 * 60 * 60));
   160→  const diffDays = Math.floor(diffMs / (1000 * 60 * 60 * 24));
   161→
   162→  if (diffMinutes < 1) return 'Baru saja';
   163→  if (diffMinutes < 60) return `${diffMinutes} menit yang lalu`;
   164→  if (diffHours < 24) return `${diffHours} jam yang lalu`;
   165→  return `${diffDays} hari yang lalu`;
   166→}

   168→function isOnline(dateStr: string): boolean {
   169→  const now = new Date();
   170→  const lastActive = new Date(dateStr);
   171→  const diffMinutes = Math.floor((now.getTime() - lastActive.getTime()) / (1000 * 60));
   172→  return diffMinutes < 5;
   173→}
